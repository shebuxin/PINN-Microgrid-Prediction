\section*{1. Where PINNs come from}

\begin{frame}{The Raissi 2019 idea, in one slide}
\textbf{Classical setting:} solve a partial differential equation
\begin{equation*}
\mathcal{N}[u](x, t) = 0, \qquad u(x, 0) = u_0(x), \qquad u(x, t)|_{\partial \Omega} = g
\end{equation*}
by training a neural net $u_\theta(x,t)$ with the loss
\begin{equation*}
\boxed{\;\mathcal{L} \;=\; \underbrace{\tfrac{1}{N_d}\sum_i \|u_\theta(x_i, t_i) - u_i\|^2}_{\text{data fit}} \;+\; \lambda \cdot \underbrace{\tfrac{1}{N_c}\sum_j \|\mathcal{N}[u_\theta](x_j, t_j)\|^2}_{\text{physics residual at collocation points}}\;}
\end{equation*}
\vspace{0.2em}
\begin{itemize}
  \item Data points $x_i, t_i$ where we measure $u$ -- typically few.
  \item Collocation points $x_j, t_j$ where the PDE \emph{should} hold -- can be cheap to sample on a grid.
  \item Net learns to fit data \emph{and} respect the equation.
\end{itemize}
\vspace{0.3em}
\begin{block}{Why this is powerful}
The physics term gives the model a strong inductive prior. It often extrapolates outside the data region in a way a pure data-fit net cannot.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{Three reasons to put physics in the loss}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{1. Sample efficiency}
\begin{itemize}
  \item PDE residual is a ``free supervisor'' anywhere we can evaluate it.
  \item Useful when labeled data is expensive (e.g.\ instrumentation, expert labels).
\end{itemize}
\vspace{0.4em}
\textbf{2. Extrapolation}
\begin{itemize}
  \item A plain net is reliable only where it has data.
  \item Physics terms constrain the model in unmeasured regions.
\end{itemize}
\column{0.5\textwidth}
\textbf{3. Trustworthiness}
\begin{itemize}
  \item Hard for a non-expert to audit a black-box prediction.
  \item Easier to defend a model that ``never violates Ohm's law'' or ``always respects voltage limits''.
\end{itemize}
\vspace{0.4em}
\begin{alertblock}{For safety screening}
Reasons 1 and 3 dominate. The project has only 77 labeled samples (data is expensive) and is for a safety-critical task (auditability matters). Both push toward physics-informed methods.
\end{alertblock}
\end{columns}
\end{frame}

\section*{2. Loss design}

\begin{frame}{The general data + physics decomposition}
\begin{equation*}
\boxed{\;\mathcal{L}_{\text{total}} \;=\; \mathcal{L}_{\text{data}} \;+\; \lambda \cdot \mathcal{L}_{\text{phys}}\;}
\end{equation*}
\vspace{0.2em}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Data loss}
\begin{itemize}
  \item BCE for classification.
  \item MSE for regression.
  \item Cross-entropy for multi-class.
  \item Standard ML, evaluated on labeled training samples.
\end{itemize}
\column{0.5\textwidth}
\textbf{Physics loss}
\begin{itemize}
  \item PDE residual (Raissi-style).
  \item Engineering limit violation (project-style).
  \item Conservation law residual.
  \item Symmetry / invariance penalty.
  \item Evaluated on \emph{any} points where physics is defined.
\end{itemize}
\end{columns}
\vspace{0.5em}
\begin{block}{Where physics points come from}
For PDE problems: sample uniformly on the domain. For our project: random scenario--contingency pairs, or grid points in feature space. They do \emph{not} need labels; the physics rule provides the supervisory signal.
\end{block}
\end{frame}

\vspace{0.4em}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Symptoms of bad $\lambda$}
\begin{itemize}
  \item Too small: physics has no effect; same as plain MLP.
  \item Too large: model ignores data, predicts whatever satisfies physics.
  \item Just right: physics regularizes without dominating.
\end{itemize}
\column{0.5\textwidth}
\textbf{Practical approach}
\begin{itemize}
  \item Start at $\lambda = 1$.
  \item Sweep over $\{0.01, 0.1, 1, 10, 100\}$ on validation.
  \item Pick the $\lambda$ that maximizes the target metric (recall, AUPRC, etc.).
\end{itemize}
\end{columns}
\end{frame}

% ============================================================
\begin{frame}{Multi-task framing}
Many real PINNs are not just one data term + one physics term -- they are several losses summed:
\begin{equation*}
\mathcal{L}_{\text{total}} \;=\; \lambda_1 \mathcal{L}_{\text{cls}} \;+\; \lambda_2 \mathcal{L}_{\text{sev}} \;+\; \lambda_3 \mathcal{L}_{\text{limit}} \;+\; \lambda_4 \mathcal{L}_{\text{rank}} \;+\; \ldots
\end{equation*}
Each term enforces a different aspect of the desired behavior. The architecture often has multiple output heads sharing a backbone.

\section*{3. Soft vs strict constraints}

\begin{frame}{Soft constraints (the usual PINN)}
A soft constraint is just a penalty term:
\begin{equation*}
\mathcal{L}_{\text{phys}} \;=\; \tfrac{1}{N_c} \sum_j \|\, \text{constraint violation at }(x_j) \,\|^2
\end{equation*}
\vspace{0.2em}
\textbf{Example (voltage limit):} we want the model to predict ``unsafe'' whenever $|V_a| < 0.95$. Define the penalty:
\begin{equation*}
\mathcal{L}_{\text{V-limit}} \;=\; \tfrac{1}{N_c} \sum_j \bigl[\, \mathbb{I}(V_j < 0.95) \cdot \max(0, \; 0.5 - \hat{y}_\theta(x_j))^2 \bigr]
\end{equation*}
Zero when the model already predicts $\hat{y} \geq 0.5$ (unsafe) on those points. Quadratic and positive otherwise.
\vspace{0.4em}
\begin{block}{Properties}
\begin{itemize}
  \item Easy to implement: just add a term to the loss.
  \item Gradient-friendly: backprop works as usual.
  \item Model \emph{can} still violate the constraint -- the penalty just makes it costly.
\end{itemize}
\end{block}
\end{frame}

% ============================================================
\begin{frame}{Strict constraints (projection or architecture)}
A strict constraint cannot be violated:
\begin{itemize}
  \item \textbf{Projection:} after the model's forward pass, project the prediction onto the feasible set. E.g., for power flow, clamp predicted voltages to a known band.
  \item \textbf{Architecture:} design the network so its outputs structurally satisfy the constraint. E.g., output magnitudes via a softplus to guarantee positivity; output a distribution via softmax to guarantee summation to 1.
  \item \textbf{Lagrangian / augmented:} explicit equality constraints with Lagrange multipliers, updated during training.
\end{itemize}
\vspace{0.5em}
\begin{block}{Trade-offs}
Strict constraints are necessary when violations are catastrophic and there's no downstream safety net. They are harder to design and may slow training. \textbf{For a screening model with a power-flow solver fallback, soft constraints are usually enough.}
\end{block}
\end{frame}

\begin{block}{Why soft constraints are sufficient here}
The model is a \emph{pre-screener}. Anything it flags as suspicious is re-checked by the exact power-flow solver. So a small violation of the soft constraint at inference time does no harm -- the solver catches it. Strict constraints would matter only if we removed the solver fallback.
\end{block}
\begin{alertblock}{What changes for a deployed safety controller}
If the model directly actuates an inverter or relay, we want hard constraints, conformal prediction, or formal verification. That's out of scope for screening but on the research roadmap.
\end{alertblock}

\section*{4. The project's limit-aware flavor}

\begin{frame}{What ``physics-informed'' means in this project}
The microgrid project does \emph{not} encode the full AC power-flow residual into the loss. Instead, it uses \textbf{five engineering-limit penalties}, each on a quantity that the simulator already gives us.
\vspace{0.4em}
\centering
\small
\begin{tabular}{lll}
\toprule
Term & Quantity & Penalty form \\
\midrule
$\mathcal{L}_{\text{V-lim}}$ & per-phase voltage $V_{i,\phi}$ vs $[0.95, 1.05]$ & $\max(0, V_{\min} - V)^2 + \max(0, V - V_{\max})^2$ \\
$\mathcal{L}_{\text{I-lim}}$ & per-phase loading $\ell_\phi$ vs 100\% & $\max(0, \ell - 1.0)^2$ \\
$\mathcal{L}_{\text{VUF}}$ & voltage unbalance vs 2\% & $\max(0, \text{VUF} - 0.02)^2$ \\
$\mathcal{L}_{\text{cls}}$ & probability head $\hat{p}$ vs label & weighted BCE \\
$\mathcal{L}_{\text{sev}}$ & severity head $\hat{s}$ vs severity score & MSE \\
\bottomrule
\end{tabular}
\vspace{0.5em}
\begin{alertblock}{Honest framing}
This is \emph{limit-aware} learning rather than strict Raissi-PINN. The same name (``physics-informed'') has stretched to cover both. State which flavor you mean in your write-up so readers know what to expect.
\end{alertblock}
\end{frame}

% ============================================================
\begin{frame}{Severity regression as physics-informed}
Instead of only predicting a binary label, also predict a \emph{continuous severity score}:
\begin{equation*}
s = \max\!\left(\, \tfrac{0.95 - V_{\min}}{0.05},\; \tfrac{\ell_{\max} - 1.0}{0.5},\; \tfrac{\text{VUF} - 0.02}{0.02} \,\right) \cdot \mathbb{I}(\text{any limit crossed})
\end{equation*}
\vspace{0.2em}
\begin{itemize}
  \item Larger $s$ means worse violation; $s = 0$ means safe.
  \item The severity head learns to predict $s$ via MSE loss.
  \item The classification head and severity head share the backbone, so the two tasks help each other.
\end{itemize}
\vspace{0.4em}
\begin{block}{Why this helps}
A single binary label compresses a lot of information into one bit. Severity gives the network a much richer supervisory signal -- specifically a \emph{continuous physical quantity}. The classification head benefits from learning the severity geometry alongside.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{Ranking loss for top-$k$ screening}
For pre-screening, the \emph{order} of risk scores matters more than absolute probabilities. We may not have well-calibrated probabilities, but we want unsafe scenarios to be at the top of the queue.
\vspace{0.4em}
\begin{equation*}
\mathcal{L}_{\text{rank}} \;=\; \tfrac{1}{|\mathcal{P}|} \sum_{(i, j) \in \mathcal{P}} \max\!\bigl(0, \;m - (\hat{y}_i - \hat{y}_j)\bigr)
\end{equation*}
where $\mathcal{P} = \{(i, j) : y_i = 1, y_j = 0\}$ is the set of (unsafe, safe) pairs and $m$ is a margin (e.g.\ $m=0.2$).
\vspace{0.3em}
\begin{itemize}
  \item Pairwise hinge: penalizes pairs where the unsafe scenario has a \emph{lower} score than a safe one.
  \item Pushes unsafe scores above safe scores by at least margin $m$.
  \item Improves top-$k$ hit rate, which is the project's actual operational metric.
\end{itemize}
\vspace{0.3em}
\begin{block}{Why this is ``physics-informed'' in spirit}
The desired ordering (\emph{more dangerous on top}) is itself a physical property of the system, not just a statistical convenience. Encoding it explicitly is a form of inductive bias.
\end{block}
\end{frame}

\section*{5. Worked example -- regularizing an MLP with safety rules}

\textbf{Training data.} Only 30 labeled scenarios, clustered near the boundary at $(V, \ell) \in [0.93, 1.02] \times [0.85, 1.05]$. \emph{No labels in the far corners.}

\end{block}

% ============================================================
\begin{frame}{Adding a limit-aware physics penalty}
\textbf{Physics rule.} At any feature-space point $(V, \ell)$:
\begin{itemize}
  \item if $V < 0.95$ or $\ell > 1.0$, the prediction should be $\hat{y} \geq 0.5$ (unsafe).
  \item if $V \geq 0.95$ and $\ell \leq 1.0$, the prediction should be $\hat{y} < 0.5$ (safe).
\end{itemize}
\textbf{Physics loss.} Sample $M = 1000$ random points $(V_j, \ell_j)$ uniformly over $[0.85, 1.10] \times [0.60, 1.30]$ -- no labels needed -- and add:
\begin{align*}
\mathcal{L}_{\text{phys}} \;=\; \tfrac{1}{M} \sum_j \biggl[
& \mathbb{I}(V_j < 0.95 \;\text{or}\; \ell_j > 1.0) \cdot \max(0,\; 0.5 - \hat{y}_\theta(V_j, \ell_j))^2 \\
+\; & \mathbb{I}(V_j \geq 0.95 \;\text{and}\; \ell_j \leq 1.0) \cdot \max(0,\; \hat{y}_\theta(V_j, \ell_j) - 0.5)^2
\biggr]
\end{align*}
\vspace{0.2em}
\textbf{Total loss.} $\mathcal{L}_{\text{total}} = \mathcal{L}_{\text{BCE on 30 labeled pts}} + \lambda \cdot \mathcal{L}_{\text{phys}}$, with $\lambda = 1.0$.
\vspace{0.4em}
\begin{alertblock}{Where the ``physics'' is}
The rule (``V or current outside limits $\Rightarrow$ unsafe'') comes from engineering knowledge, not from the 30 labels. The penalty term injects this knowledge anywhere in the feature space, even where data is absent.
\end{alertblock}
\end{frame}

\begin{frame}[fragile]{In code -- adding a physics loss to a PyTorch training loop}
\begin{lstlisting}[language=Python]
import torch, torch.nn as nn

# Plain MLP as in L05
model = nn.Sequential(nn.Linear(2, 32), nn.ReLU(),
                      nn.Linear(32, 32), nn.ReLU(),
                      nn.Linear(32, 1),  nn.Sigmoid())
opt = torch.optim.Adam(model.parameters(), lr=1e-3)
bce = nn.BCELoss()

def physics_loss(x, y_hat):
    V, I = x[:, 0], x[:, 1]
    should_be_unsafe = ((V < 0.95) | (I > 1.0)).float().unsqueeze(1)
    should_be_safe   = 1.0 - should_be_unsafe
    pen_unsafe = torch.relu(0.5 - y_hat) ** 2 * should_be_unsafe
    pen_safe   = torch.relu(y_hat - 0.5) ** 2 * should_be_safe
    return (pen_unsafe + pen_safe).mean()

for epoch in range(200):
    # data loss on 30 labeled points (X_tr, y_tr)
    loss_data = bce(model(X_tr), y_tr)
    # physics loss on 1000 random unlabeled points (X_phys)
    loss_phys = physics_loss(X_phys, model(X_phys))
    loss = loss_data + 1.0 * loss_phys
    opt.zero_grad(); loss.backward(); opt.step()
\end{lstlisting}
\end{frame}

\section*{6. When PINN helps -- and when not}

\begin{frame}{When PINN helps}
\begin{itemize}
  \item \textbf{Sparse data.} Few labels but plenty of room to evaluate the physics. Most successful PINN applications fall here.
  \item \textbf{Known physics.} The constraint is correct and easy to evaluate. Voltage limits and Ohm's law qualify; ``a wise human would say this is unsafe'' does not.
  \item \textbf{Extrapolation matters.} The model will be queried outside the training distribution. Physics provides a fallback.
  \item \textbf{Safety-critical applications.} You need to defend the model's behavior to regulators, operators, or reviewers.
\end{itemize}
\vspace{0.5em}
\begin{block}{Why the project qualifies for all four}
77 labeled samples, well-defined voltage / current / VUF limits, N-1 contingency analysis is inherently out-of-distribution (different from the base case), and the application is grid security.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{When PINN does not help (or hurts)}
\begin{enumerate}
  \item \warnitem \textbf{Plenty of data, well-tuned baseline.} If a plain MLP already gets 99\% recall on a labeled dataset of millions, the physics term is noise. Common finding in large-scale tabular benchmarks.
  \item \warnitem \textbf{Wrong physics.} If the constraint is a stale or approximate rule, it can pull the model away from the true distribution. Audit the constraint before trusting it.
  \item \warnitem \textbf{Overweighted $\lambda$.} The model fits physics perfectly and ignores data. Showing up as ``training BCE never drops'' in your logs.
  \item \warnitem \textbf{Optimization landscape conflicts.} Two competing terms with mismatched gradient magnitudes can stall training. Krishnapriyan et al.\ 2021 documents many failure modes.
  \item \warnitem \textbf{Treating it as magic.} PINNs need just as much hyperparameter tuning, just as much validation discipline, as plain MLPs.
\end{enumerate}
\end{frame}

\section*{7. Project: Week 6 PI-MLP composite loss}

\begin{frame}{The five losses in Week 6}
\begin{equation*}
\mathcal{L}_{\text{PI-MLP}} \;=\; \lambda_1 \mathcal{L}_{\text{cls}} + \lambda_2 \mathcal{L}_{\text{sev}} + \lambda_3 \mathcal{L}_{\text{margin}} + \lambda_4 \mathcal{L}_{\text{consistency}} + \lambda_5 \mathcal{L}_{\text{rank}}
\end{equation*}
\vspace{0.3em}
\centering
\small
\begin{tabular}{lll}
\toprule
Term & Role & Form \\
\midrule
$\mathcal{L}_{\text{cls}}$ & main classifier & weighted BCE \\
$\mathcal{L}_{\text{sev}}$ & continuous severity signal & MSE on severity head \\
$\mathcal{L}_{\text{margin}}$ & per-component (V, I, VUF) limit penalty & hinge-squared limit violations \\
$\mathcal{L}_{\text{consistency}}$ & classification and severity agree & high $\hat{p}$ $\Leftrightarrow$ high $\hat{s}$ \\
$\mathcal{L}_{\text{rank}}$ & top-$k$ screening order & pairwise hinge \\
\bottomrule
\end{tabular}
\vspace{0.6em}
\begin{block}{Default weights used in the notebook}
$\lambda_1 = 1.0$, $\lambda_2 = 0.5$, $\lambda_3 = 0.3$, $\lambda_4 = 0.2$, $\lambda_5 = 0.3$. Tuned coarsely on validation -- the notebook walks through the sweep.
\end{block}
\end{frame}
