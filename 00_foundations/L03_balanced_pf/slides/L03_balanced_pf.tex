\vspace{0.4em}
\begin{block}{Two equivalent statements of ``balanced''}
\begin{itemize}
  \item \textbf{Time domain:} three sinusoids of equal amplitude, $120^\circ$ apart in phase.
  \item \textbf{Phasor domain:} three vectors of equal length, $120^\circ$ apart in angle.
\end{itemize}
Both views describe the same physical assumption.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{Algebraic statement: the $a$-operator}
Define the rotation operator
\begin{equation*}
a \;=\; e^{j 2\pi/3} \;=\; e^{j 120^\circ} \;=\; -\tfrac{1}{2} + j\tfrac{\sqrt{3}}{2}.
\end{equation*}
Then a balanced phase-$a$, phase-$b$, phase-$c$ set satisfies
\begin{equation*}
\boxed{\;\;\mathbf{V}_b \;=\; a^2 \, \mathbf{V}_a, \qquad \mathbf{V}_c \;=\; a \, \mathbf{V}_a\;\;}
\end{equation*}
\vspace{0.2em}
And likewise for currents:
\begin{equation*}
\mathbf{I}_b \;=\; a^2 \, \mathbf{I}_a, \qquad \mathbf{I}_c \;=\; a \, \mathbf{I}_a.
\end{equation*}
\vspace{0.3em}
\begin{alertblock}{Why this matters for computation}
If $\mathbf{V}_a$ is known, $\mathbf{V}_b$ and $\mathbf{V}_c$ are determined by multiplication. \textbf{One unknown per bus instead of three.} The whole simplification of single-line analysis flows from this identity.
\end{alertblock}
\end{frame}

% ============================================================
\begin{frame}{The three places where symmetry must hold}
\begin{columns}[T]
\column{0.33\textwidth}
\textbf{1. Source symmetry}
\begin{itemize}
  \item Generators produce three phases equal in magnitude and $120^\circ$ apart.
  \item Transformer windings are symmetric.
  \item The slack bus is itself balanced.
\end{itemize}
\column{0.33\textwidth}
\textbf{2. Load symmetry}
\begin{itemize}
  \item Every three-phase load draws equal $P$, $Q$ on each phase.
  \item No single-phase laterals.
  \item No phase-A-only PV.
\end{itemize}
\column{0.33\textwidth}
\textbf{3. Network symmetry}
\begin{itemize}
  \item Each line has equal self-impedance on all phases.
  \item Mutual coupling is the same A-B, B-C, A-C.
  \item Transposed lines or equivalent.
\end{itemize}
\end{columns}
\vspace{0.8em}
\begin{block}{All three or none}
The single-line simplification requires \emph{all three} assumptions simultaneously. Break any one and the per-phase quantities drift apart -- exactly what L04 will measure.
\end{block}
\end{frame}

\end{block}
\end{frame}

\section*{2. The single-phase equivalent}

\begin{frame}{Three coupled equations collapse to one}
Without balance, the three-phase nodal equation at each bus is
\begin{equation*}
\begin{bmatrix} \mathbf{I}_a \\ \mathbf{I}_b \\ \mathbf{I}_c \end{bmatrix}
\;=\;
\underbrace{\begin{bmatrix}
Y_{aa} & Y_{ab} & Y_{ac} \\
Y_{ba} & Y_{bb} & Y_{bc} \\
Y_{ca} & Y_{cb} & Y_{cc}
\end{bmatrix}}_{\text{3$\times$3 phase-domain admittance}}
\begin{bmatrix} \mathbf{V}_a \\ \mathbf{V}_b \\ \mathbf{V}_c \end{bmatrix}
\end{equation*}
For $N$ buses this is a $3N$-dimensional linear system.
\vspace{0.3em}

Under the three symmetry assumptions, the $3\times 3$ phase block has the structure
$Y_{aa}=Y_{bb}=Y_{cc}$ (self) and $Y_{ab}=Y_{bc}=Y_{ca}$ (mutual). Combined with the rotation identities $\mathbf{V}_b = a^2 \mathbf{V}_a$ and $\mathbf{V}_c = a \mathbf{V}_a$, the three rows above reduce to a \emph{single} equation per bus:
\begin{equation*}
\boxed{\;\;\mathbf{I}_a \;=\; Y^{(1)} \, \mathbf{V}_a\;\;}
\end{equation*}
where $Y^{(1)} = Y_{aa} + a^2 \, Y_{ab} + a \, Y_{ac}$ is the \emph{positive-sequence admittance} for $V_b=a^2V_a$, $V_c=aV_a$.
\end{frame}

  \item \textbf{Cognition:} you can read a one-line diagram on one piece of paper.
\end{itemize}
\end{block}

\begin{itemize}

  \item Wrote one PF equation per bus.

\begin{itemize}
  \item Building the \emph{positive-sequence} $Y_{\text{bus}}$.
  \item Solving the \emph{positive-sequence} PF.
  \item Reading the \emph{positive-sequence} bus voltages.
  \item Implicitly relying on all three symmetry assumptions.
\end{itemize}
\vspace{0.4em}
\begin{alertblock}{Why this re-labeling matters}
When we get to L04, ``positive sequence'' is one of three coordinates we will track. The other two -- negative and zero sequence -- are exactly the modes that single-line analysis throws away.
\end{alertblock}
\end{frame}

\section*{3. Positive sequence -- a preview of L04}

\begin{frame}{Three sequence components in one slide}
Any three-phase phasor set $(\mathbf{V}_a, \mathbf{V}_b, \mathbf{V}_c)$ can be uniquely written as the sum of three balanced sets:
\begin{align*}
\mathbf{V}_a &= \mathbf{V}^{(1)} \;+\; \mathbf{V}^{(2)} \;+\; \mathbf{V}^{(0)} \\
\mathbf{V}_b &= a^2\, \mathbf{V}^{(1)} \;+\; a\, \mathbf{V}^{(2)} \;+\; \mathbf{V}^{(0)} \\
\mathbf{V}_c &= a\, \mathbf{V}^{(1)} \;+\; a^2\, \mathbf{V}^{(2)} \;+\; \mathbf{V}^{(0)}
\end{align*}
\centering
\small
\begin{tabular}{lll}
\toprule
Component & Pattern & Meaning \\
\midrule
\textbf{Positive} ($V^{(1)}$) & balanced, rotating ABC & true balanced operation \\
\textbf{Negative} ($V^{(2)}$) & balanced, rotating ACB & a measure of phase asymmetry \\
\textbf{Zero} ($V^{(0)}$) & all three in phase & a measure of common-mode shift \\
\bottomrule
\end{tabular}
\vspace{0.3em}
\begin{block}{Key fact for this lecture}
Under balanced operation, $\mathbf{V}^{(2)} = \mathbf{V}^{(0)} = 0$. \textbf{Balanced PF is positive-sequence PF.} L04 will build all three.
\end{block}
\end{frame}

\end{block}

\section*{4. When balanced PF is appropriate}

\begin{frame}{Where balanced PF actually works well}
\textbf{Strong-balance domains:}
\begin{itemize}
  \item \textbf{HV transmission} (110--765 kV): transposed lines, three-phase generators, large aggregated loads. Phase-to-phase mismatch typically $< 0.5\%$.
  \item \textbf{Power plants and substations}: three-phase machines, transformers, breakers -- inherently symmetric.
  \item \textbf{MV industrial loads}: large motors and arc furnaces, drawing all three phases.
  \item \textbf{Bulk planning studies}: market clearing, OPF, long-term expansion -- phase detail is not the question being asked.
\end{itemize}
\vspace{0.4em}
\begin{block}{Why we still use balanced PF every day}
$\sim 3\times$ fewer variables, $\sim 9\times$ smaller admittance matrix, much faster convergence. For most transmission-scale studies, the precision is more than enough.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{How to judge whether balance is adequate}
\centering
\small
\begin{tabular}{p{0.42\textwidth}p{0.22\textwidth}p{0.26\textwidth}}
\toprule
Scenario & Expected phase asymmetry & Recommended model \\
\midrule
Transposed network with balanced three-phase devices & low & balanced PF is usually appropriate \\
Planning study using aggregated phase totals & study-dependent & compare against a 3-phase benchmark \\
LV feeder with single-phase loads, PV, or EV & material & unbalanced 3-phase PF \\
Neutral/grounding effects or phase-specific protection & material & unbalanced 3-phase PF \\
Unbalanced faults & intrinsic & sequence/phase-domain fault model \\
\bottomrule
\end{tabular}
\vspace{0.6em}
\begin{block}{Practical recipe}
Do not assume an error percentage from voltage level alone. Compare balanced and three-phase results on representative operating points, then decide whether the approximation preserves the limits relevant to the study.
\end{block}
\end{frame}

\section*{5. When balanced PF lies}

\begin{frame}{A diagnostic checklist}
\textbf{If \emph{any} of the following is true, balanced PF can hide real violations:}
\begin{enumerate}
  \item Loads or DER attached to one or two phases only.
  \item Line parameters that differ phase-to-phase (untransposed LV).
  \item Neutral conductor effects or grounding asymmetry.
  \item Fault analysis (faults are inherently unbalanced).
  \item Single-phase protection or per-phase tap controllers.
\end{enumerate}
\vspace{0.4em}
\begin{alertblock}{The microgrid baseline}
LV microgrids match items 1, 2, and often 3. Doing balanced PF in this regime and quoting per-phase results would be dishonest. We will not do that in this course.
\end{alertblock}
\end{frame}

% ============================================================
\begin{frame}{A numerical example: balanced PF lies}
A 2-bus LV feeder. Line: $z = 0.05 + j0.05$ pu (high R/X, LV-like). Slack at $V_1 = 1.00$.\\[0.3em]
Total load at bus 2 is the same in both scenarios:
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Scenario A: balanced load}
\begin{itemize}
  \item $0.133$ pu on each of phases A, B, C
  \item Total $P = 0.4$ pu, balanced
\end{itemize}
\column{0.5\textwidth}
\textbf{Scenario B: same total, all on phase A}
\begin{itemize}
  \item $0.4$ pu on phase A, $0$ on B and C
  \item Total $P = 0.4$ pu, very unbalanced
\end{itemize}
\end{columns}
\vspace{0.6em}
\centering
\small
\begin{tabular}{lccc}
\toprule
 & $|V_a|$ at bus 2 & $|V_b|$ at bus 2 & $|V_c|$ at bus 2 \\
\midrule
\textbf{What balanced PF reports} (both scenarios) & 0.979 & 0.979 & 0.979 \\
\textbf{What 3$\phi$ PF says, Scenario A} (truly balanced) & 0.979 & 0.979 & 0.979 \\
\textbf{What 3$\phi$ PF says, Scenario B} (load on A only) & \textbf{0.908} & 1.029 & 0.995 \\
\bottomrule
\end{tabular}
\vspace{0.5em}
\begin{alertblock}{The lie}
For Scenario B, balanced PF reports ``$V = 0.979$, no violation anywhere''. The true phase-A voltage is $0.908$ -- well below the $0.95$ limit. \textbf{Phase A is in violation and the balanced model cannot see it.} L04 will derive these numbers from scratch using symmetrical components.
\end{alertblock}
\end{frame}

\section*{6. The decision: balanced or three-phase?}

\node[q] (q1) {Are loads, DER, and lines all symmetric on the three phases?};
\node[a, fill=green!12, below left=of q1] (yes) {YES \\ use balanced PF (\code{runpp})};
\node[a, fill=red!12, below right=of q1] (no)  {NO \\ use three-phase PF (\code{runpp\_3ph})};
\draw[->, thick] (q1.south) -- node[left, font=\tiny, xshift=-0.3cm]{yes} (yes.north);
\draw[->, thick] (q1.south) -- node[right, font=\tiny, xshift=0.3cm]{no} (no.north);
\end{tikzpicture}
\vspace{0.6em}
\begin{block}{Decision in plain language}
If you can answer ``yes'' confidently and the cost of being wrong is low (planning, market clearing), use balanced PF. If you cannot, or if a hidden violation is consequential (operations, safety screening), pay the extra cost of \code{runpp\_3ph}.
\end{block}
\end{frame}
