% ============================================================
\begin{frame}{Why an MLP on flattened features falls short}
Naive approach: take every bus's features, stack into one long vector, feed to an MLP.
\begin{equation*}
\mathbf{x}_{\text{flat}} = \bigl[\, h_1 \mid h_2 \mid \ldots \mid h_N \,\bigr] \in \mathbb{R}^{N \cdot d}, \qquad \hat{y} = \text{MLP}(\mathbf{x}_{\text{flat}})
\end{equation*}
\vspace{0.2em}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{What the MLP cannot do}
\begin{itemize}
  \item It does not know that bus 3 is adjacent to bus 4 but not bus 1.
  \item It cannot tell a 5-bus feeder from a 10-bus feeder (input size differs).
  \item It must \emph{relearn} the relationship between adjacent buses from data.
  \item Heavy permutation sensitivity: shuffling the bus order gives a different prediction.
\end{itemize}
\column{0.5\textwidth}
\begin{block}{What we want}
A model that takes a \emph{graph} as input directly, respects the topology, and works on any size / shape of network. That is a Graph Neural Network.
\end{block}
\begin{alertblock}{Project's honest baseline}
On 77 samples, a flat-MLP often \emph{wins} simply because there isn't enough data to use the topology. We still build the GNN -- it's the right framework for scaling up.
\end{alertblock}
\end{columns}
\end{frame}

\section*{2. Message passing}

\begin{frame}{Message passing in three steps}
\centering
\begin{tikzpicture}[node distance=0.4cm and 0.6cm, every node/.style={font=\scriptsize}]
\tikzset{box/.style={draw, rounded corners, align=center, minimum width=2.6cm, minimum height=1.0cm}}
\tikzset{arr/.style={-{Latex[length=2.5mm]}, thick}}
\node[box, fill=blue!10] (a) {1. \textbf{Aggregate}\\sum / mean / max\\of neighbours};
\node[box, fill=warnorange!18, right=of a] (b) {2. \textbf{Update}\\combine with self\\(linear $+$ ReLU)};
\node[box, fill=darkgreen!18, right=of b] (c) {3. \textbf{Repeat}\\next layer pulls\\in 2-hop info};
\draw[arr] (a) -- (b);
\draw[arr] (b) -- (c);
\draw[arr, gray] (c.south) to[bend left=30] node[below, font=\tiny]{next layer} (a.south);
\end{tikzpicture}
\vspace{0.6em}
\textbf{In words.} At each layer:
\begin{enumerate}
  \item Each node \emph{collects} feature vectors from its neighbours.
  \item It \emph{combines} them with its own current state via a small learned function (typically: linear layer $+$ nonlinearity).
  \item After $L$ layers, every node has aggregated information from its $L$-hop neighbourhood.
\end{enumerate}
\vspace{0.4em}
\begin{block}{The key insight}
A GNN is just an MLP with a particular weight-tying pattern: \emph{the same small network is applied at every node, using its local neighbourhood as input.} Topology comes in through the neighbour list.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{The general message-passing equation}
For each node $v$ at layer $l$:
\begin{equation*}
\boxed{\;h_v^{(l+1)} \;=\; \text{UPDATE}\!\left( h_v^{(l)},\; \text{AGG}\bigl(\{ h_u^{(l)} : u \in \mathcal{N}(v)\}\bigr) \right)\;}
\end{equation*}
\vspace{0.2em}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Common choices}
\begin{itemize}
  \item \textbf{AGG}: sum, mean, max, attention-weighted sum.
  \item \textbf{UPDATE}: linear $+$ ReLU, GRU cell, MLP.
\end{itemize}
\textbf{Important properties}
\begin{itemize}
  \item AGG must be \emph{permutation-invariant} (no ``first neighbour'').
  \item All nodes share the same weights at a given layer (parameter sharing).
\end{itemize}
\column{0.42\textwidth}
\begin{block}{Different choices = different GNNs}
The next three slides each pick a specific AGG and UPDATE: GCN, GraphSAGE, and GAT. They differ in expressivity and in how they handle different neighbourhood sizes.
\end{block}
\end{columns}
\end{frame}

\section*{3. Major variants}

\begin{frame}{GCN -- Graph Convolutional Network (Kipf \& Welling, 2017)}
The cleanest baseline. Layer-wise update:
\begin{equation*}
H^{(l+1)} \;=\; \sigma\!\left( \tilde{D}^{-1/2} \tilde{A} \tilde{D}^{-1/2} \, H^{(l)} \, W^{(l)} \right)
\end{equation*}
where
\begin{itemize}
  \item $\tilde{A} = A + I$ is the adjacency matrix with self-loops (so each node aggregates itself too).
  \item $\tilde{D}$ is the diagonal degree matrix of $\tilde{A}$.
  \item $W^{(l)}$ is a learnable weight matrix (the same MLP layer applied at every node).
\end{itemize}
\vspace{0.4em}
\begin{block}{Per-node view}
\begin{equation*}
h_v^{(l+1)} = \sigma\!\Bigl(\, W^{(l)} \cdot \sum_{u \in \mathcal{N}(v) \cup \{v\}} \frac{1}{\sqrt{\deg(v)+1}\sqrt{\deg(u)+1}} \, h_u^{(l)} \Bigr)
\end{equation*}
A weighted mean of self and neighbours, then linear projection, then nonlinearity. The normalization keeps activations stable.
\end{block}
\end{frame}

% ============================================================
\begin{frame}{GraphSAGE -- learnable aggregator (Hamilton et al., 2017)}
\begin{equation*}
h_v^{(l+1)} \;=\; \sigma\!\bigl( W^{(l)} \cdot [\, h_v^{(l)}, \;\text{AGG}(\{ h_u^{(l)} : u \in \mathcal{N}(v)\}) \,]\bigr)
\end{equation*}
\vspace{0.2em}
\textbf{Key differences from GCN:}
\begin{itemize}
  \item Concatenate self and aggregated neighbours (not sum them).
  \item AGG is one of: mean, max-pool, sum, LSTM.
  \item No global normalization. Easier to apply to graphs of varying size.
\end{itemize}
\vspace{0.4em}
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Inductive}
\begin{itemize}
  \item Trained on one graph, can be applied to \emph{unseen} graphs.
  \item Useful when test feeders differ from training feeders.
\end{itemize}
\column{0.5\textwidth}
\textbf{Practical}
\begin{itemize}
  \item Mean or max-pool aggregator is fast and competitive.
  \item LSTM aggregator is more expressive but slower.
\end{itemize}
\end{columns}
\end{frame}

% ============================================================
\begin{frame}{GAT -- Graph Attention Network (Veli\v{c}kovi\'c et al., 2018)}
\begin{equation*}
h_v^{(l+1)} \;=\; \sigma\!\Bigl( \sum_{u \in \mathcal{N}(v) \cup \{v\}} \alpha_{vu} \, W^{(l)} \, h_u^{(l)} \Bigr)
\end{equation*}
where $\alpha_{vu}$ is a \emph{learned attention coefficient}:
\begin{equation*}
\alpha_{vu} \;=\; \text{softmax}_u \bigl( \text{LeakyReLU}( a^\top [W h_v \,|\, W h_u]) \bigr)
\end{equation*}
\vspace{0.3em}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{What this buys}
\begin{itemize}
  \item Each neighbour gets a learned importance weight.
  \item Multi-head attention (run $K$ in parallel, concatenate) increases expressivity.
  \item Works well when some neighbours matter much more than others (e.g., a heavily-loaded line).
\end{itemize}
\column{0.42\textwidth}
\begin{block}{Cost}
More parameters, more computation, more risk of overfitting on small graphs. Use when GCN / SAGE underperform and you have the data to support it.
\end{block}
\end{columns}
\end{frame}

% ============================================================
\begin{frame}{Pooling to graph-level}
After $L$ layers we have an embedding $h_v$ at every node. To produce a single graph-level prediction:
\begin{equation*}
\mathbf{z}_G \;=\; \text{POOL}(\{h_v : v \in V\}), \qquad \hat{y} \;=\; \text{MLP}(\mathbf{z}_G)
\end{equation*}
\vspace{0.2em}
Common POOL choices:
\begin{itemize}
  \item \textbf{mean pool}: $\mathbf{z}_G = \tfrac{1}{|V|} \sum_v h_v$ -- bias toward typical bus state.
  \item \textbf{max pool}: $\mathbf{z}_G = \max_v h_v$ (elementwise) -- bias toward worst-case bus.
  \item \textbf{sum pool}: $\mathbf{z}_G = \sum_v h_v$ -- preserves overall magnitude.
  \item \textbf{attention pool}: learned weights $\alpha_v$ over nodes.
\end{itemize}
\vspace{0.3em}
\begin{alertblock}{For safety screening, max-pool is a sensible default}
The graph is unsafe \emph{if any one bus} crosses a limit. Max-pool naturally captures this ``worst case wins'' logic. Mean-pool can hide one bad bus by averaging it with many good ones.
\end{alertblock}
\end{frame}

\begin{block}{Strategy}

\end{block}

% ============================================================
\begin{frame}{One GCN layer, by hand}
Using the simplified mean aggregator (close enough to GCN for intuition):
\begin{equation*}
h_v^{(1)} \;=\; \sigma\!\Bigl(\, W \cdot \frac{1}{|\mathcal{N}(v)| + 1} \sum_{u \in \mathcal{N}(v) \cup \{v\}} h_u^{(0)} \Bigr)
\end{equation*}
\vspace{0.2em}
For node 2, $\mathcal{N}(2) = \{1, 3\}$. The aggregated input (mean of itself, node 1, node 3) is:
\begin{equation*}
\bar{h}_2 = \tfrac{1}{3}\bigl( (0.98, 0.10, 0.04) + (1.00, 0.0, 0.0) + (0.97, 0.15, 0.06) \bigr) = (0.983,\; 0.083,\; 0.033)
\end{equation*}
The learnable matrix $W$ then maps this 3-vector to a hidden representation (e.g., 16-dim), and $\sigma = $ ReLU.
\vspace{0.4em}
\begin{block}{What just happened}
Node 2's new embedding $h_2^{(1)}$ \emph{depends on its neighbours' features}. After one layer it knows about nodes 1 and 3 (1-hop neighbourhood). After two layers it will also know about node 4 (via node 3).
\end{block}
\end{frame}

\end{tikzpicture}
\vspace{0.5em}
\begin{columns}[T]
\column{0.55\textwidth}
\textbf{End-to-end forward pass}
\begin{itemize}
  \item Node features in $\mathbb{R}^{N \times 3}$.
  \item After 2 GCN layers: $\mathbb{R}^{N \times 16}$ embeddings.
  \item After max-pool over nodes: a single $\mathbb{R}^{16}$ graph vector.
  \item MLP head with sigmoid: scalar risk score $\hat{y} \in [0,1]$.
\end{itemize}
\column{0.42\textwidth}
\begin{block}{Training}
Same as L05: minibatch of (graph, label) pairs, BCE loss, Adam, early stop on validation. Use class-weighted BCE for the project's $47/30$ imbalance.
\end{block}
\end{columns}
\end{frame}

\begin{frame}[fragile]{In code: plain PyTorch vs PyTorch Geometric}
\begin{lstlisting}[language=Python]
# Option A: PyTorch Geometric (production standard)
import torch.nn as nn
from torch_geometric.nn import GCNConv, global_max_pool

class GridGNN(nn.Module):
    def __init__(self, in_dim=3, hidden=16):
        super().__init__()
        self.g1   = GCNConv(in_dim,  hidden)
        self.g2   = GCNConv(hidden, hidden)
        self.head = nn.Sequential(nn.Linear(hidden, hidden), nn.ReLU(),
                                  nn.Linear(hidden, 1),     nn.Sigmoid())
    def forward(self, x, edge_index, batch):
        h = self.g1(x, edge_index).relu()
        h = self.g2(h, edge_index).relu()
        z = global_max_pool(h, batch)          # batch indicates which graph each node belongs to
        return self.head(z)
\end{lstlisting}
\begin{alertblock}{What the project actually uses}
The Week 7 notebook implements the GCN-style message passing in \emph{plain PyTorch} (no PyG dependency) so students can read every line. Same math, more lines of code, no extra install.
\end{alertblock}
\end{frame}

\node[n, fill=warnorange!22] (c) at (2,0.8) {};

\node[font=\tiny] at (1, -0.5) {all identical};
\end{scope}

\vspace{0.5em}
\begin{block}{What's happening}
Each layer mixes a node with its neighbours. After many layers, every node's embedding becomes a global average -- node identity is washed out.
\end{block}
\begin{alertblock}{Practical advice}
Limit depth to 2--4 layers for most problems. If you need deeper, use residual connections, jumping knowledge, or DropEdge.
\end{alertblock}
\end{frame}

\begin{block}{The take-away}
Message-passing GNNs are upper-bounded by the 1-WL graph isomorphism test, and particular aggregators can be less expressive. There exist non-isomorphic graphs they \emph{cannot} distinguish. Structural features or higher-order architectures can go beyond this limit.
\end{block}

\begin{columns}[T]
\column{0.55\textwidth}
\textbf{Why GNN loses here}
\begin{itemize}
  \item Only 77 labeled scenarios.
  \item GNN parameter sharing helps when graph diversity is high -- but we have one feeder.
  \item Flat MLP can memorize 77 patterns; GNN has stronger inductive bias that doesn't pay off at this scale.
\end{itemize}
\column{0.42\textwidth}
\begin{alertblock}{We still build the GNN}
The architecture is right; the dataset is too small. Scale to hundreds of feeders and the picture inverts. This is a real lesson about how AI choices depend on data regime, not just method elegance.
\end{alertblock}
\end{columns}

\begin{block}{What is ``phase-aware''}
Each bus carries \emph{three} voltage channels ($V_a, V_b, V_c$) instead of one balanced positive-sequence voltage. The same node-feature vector contains phase-A, B, C quantities together, so the GNN learns from the unbalance pattern directly.
\end{block}

\section*{8. Recap}

\begin{frame}{Homework}
\begin{block}{Required}
\begin{enumerate}
  \item Train a 2-layer GCN on the Cora citation dataset using PyTorch Geometric (or plain PyTorch). Achieve $\geq 80\%$ test accuracy on node classification.
  \item Visualize the learned node embeddings with t-SNE (or PCA) at layer 1 and layer 2. Are the classes separating?
  \item Replace GCN with GraphSAGE (mean aggregator). Compare test accuracy. Run each with 3 seeds and report mean / stdev.
  \item Implement a tiny graph-level classifier: 4-bus feeder graphs labeled by your own rule (e.g.\ unsafe if $V_{\min} < 0.95$). Train a 2-layer GCN with max-pool readout. Report recall on a held-out set.
\end{enumerate}
\end{block}
\begin{block}{Optional}
\begin{enumerate}
  \item Increase the GCN depth from 2 to 8 layers. Plot validation accuracy vs depth. Where does oversmoothing kick in?
  \item Try edge features by manually concatenating $(R, X)$ to each message. Does it help?
\end{enumerate}
\end{block}
\end{frame}
